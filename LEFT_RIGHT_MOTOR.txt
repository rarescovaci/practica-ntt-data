void setup() {
  pinMode(7, OUTPUT);
  pinMode(8, OUTPUT);
  pinMode(9, OUTPUT);
  
}

void loop() {
  digitalWrite(8,HIGH);
  digitalWrite(7, LOW);
  analogWrite(9, 250);
  delay(3000);
  analogWrite(9, 250);
    digitalWrite(8,LOW);
  digitalWrite(7, LOW);

  digitalWrite(7, HIGH);
  analogWrite(9, 250);
  delay(3000);
  analogWrite(9, 250);
     digitalWrite(8,LOW);
  digitalWrite(7, LOW);
  delay(1000);
  
}
